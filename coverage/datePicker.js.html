import React, { Component } from "react";
import ReactDOM from 'react-dom';
import _ from 'lodash';
 
import CalendarContent from './CalendarContent';
 
 
export class DatePicker extends Component{
    constructor(){
        super();
        this.state={
            inputDate:this.getToday(),
            datePicker:this.getToday(),
            calendarContentCss:'d-none'
        };
    }
 
    componentDidMount() {
        this.addEventListener();
    }
 
    addEventListener(){
        document.addEventListener('click', this.handleClickOutside.bind(this), true);
    }
 
    componentWillUnmount() {
        this.removeEventListener();
    }
 
    removeEventListener(){
        document.removeEventListener('click', this.handleClickOutside.bind(this), true);
    }
 
    handleClickOutside(event) {
 
        const domNode = ReactDOM.findDOMNode(this);
 
        if (!domNode || !domNode.contains(event.target)) {
            this.setShowCalendarContent('d-none');
        }
    }
 
 
    getToday(){
        let date = new Date();
        let year = date.getFullYear(),
            month = date.getMonth()+1,
            day = date.getDate();
 
        let today={'year':year,'month':month,'day':day};
 
        let todayFormat = this.formatCalendarToString(today);
 
 
        return _.clone(todayFormat);
    }
 
    inputDateFormat(){
        let inputDate = this.state.inputDate;
        let datePicker = this.state.datePicker;
 
        if(inputDate.year == undefined || inputDate.month== undefined || inputDate.day== undefined){
 
            let datePickerFormat = this.formatCalendarToString(datePicker);
            inputDate = _.clone(datePickerFormat);
        }
 
        let formatDate = inputDate.year+'-'+inputDate.month+'-'+inputDate.day;
 
        return formatDate;
    }
 
    handleChange(event){
 
        let inputDate = this.state.inputDate;
        let isDate = this.validateDateFormate(event.target.value);
        let date = _.split(event.target.value,'-');
        inputDate.year = date[0];
        inputDate.month = date[1];
        inputDate.day = date[2];
 
        this.setState({inputDate: inputDate });
 
        if(isDate){
            this.setState({datePicker: _.clone(inputDate) });
        }
 
    }
 
    validateDateFormate(date){
        //日期格式以及閏年判斷
        let dateReg = new RegExp("(([0-9]{3}[1-9]|[0-9]{2}[1-9][0-9]{1}|[0-9]{1}[1-9][0-9]{2}|[1-9][0-9]{3})-(((0[13578]|1[02])-(0[1-9]|[12][0-9]|3[01]))|((0[469]|11)-(0[1-9]|[12][0-9]|30))|(02-(0[1-9]|[1][0-9]|2[0-8]))))|((([0-9]{2})(0[48]|[2468][048]|[13579][26])|((0[48]|[2468][048]|[3579][26])00))-02-29)");
 
        return dateReg.test(date);
    }
 
    dateSelect(mode,eachCalendar,currentCalendar,today){
        let testCalendar= this.state.datePicker;
        let isSelect,isToday;
 
        isSelect =
            eachCalendar==testCalendar.day && currentCalendar.month==testCalendar.month && currentCalendar.year==testCalendar.year ;
 
        isToday =
            eachCalendar==today.day && currentCalendar.month==today.month && currentCalendar.year==today.year ;
 
        return {isSelect,isToday};
    }
 
    changeInputDate(currentCalendar){
 
        let currentCalendarFormat = this.formatCalendarToString(currentCalendar);
        this.setState({datePicker:_.clone(currentCalendarFormat)},()=>{
            this.setShowCalendarContent('d-none');
        });
 
    }
 
    formatCalendarToString(currentCalendar){
        let currentCalendarFormat = _.clone(currentCalendar);
 
        currentCalendarFormat.year = currentCalendarFormat.year.toString();
        currentCalendarFormat.month = currentCalendarFormat.month.toString();
        currentCalendarFormat.day = currentCalendarFormat.day.toString();
 
        if(currentCalendarFormat.year.length<4){
            let year = currentCalendarFormat.year;
            let yearMiddleToEnd = 4 - currentCalendarFormat.year.length;
            for(let counter=0;counter<yearMiddleToEnd;counter++){
                year = "0"+year;
            }
 
            currentCalendarFormat.year = year;
        }
 
 
 
        if(currentCalendarFormat.day.length==1){
            currentCalendarFormat.day ='0'+currentCalendarFormat.day;
        }
        if(currentCalendarFormat.month.length==1){
            currentCalendarFormat.month ='0'+currentCalendarFormat.month;
        }
 
        return currentCalendarFormat;
    }
 
    setCalendarContentCss(e){
        e.stopPropagation();
        this.setShowCalendarContent('');
    }
 
    setShowCalendarContent(css){
        let datePicker = this.state.datePicker;
        this.setState({inputDate:_.clone(datePicker)});
 
        this.setState({calendarContentCss: css});
    }
 
    formatDatePickerToNumber(){
        let datePicker=  _.clone(this.state.datePicker);
        datePicker.year = parseInt(datePicker.year);
        datePicker.month = parseInt(datePicker.month);
        datePicker.day = parseInt(datePicker.day);
 
        // console.log('datePicker:  ',datePicker);
 
        return datePicker;
    }
 
 
    render(){
        return(
            <div className="position-relative">
 
                <div onClick={this.setShowCalendarContent.bind(this,'d-none')}>
                    <div className="position-relative">
 
                        <input
                            type="text"
                            value={this.inputDateFormat()}
                            onChange={this.handleChange.bind(this)}
                            onClick={this.setCalendarContentCss.bind(this)}
                            className="pl-5"
                        />
                        <i className="fas fa-calendar-alt position-absolute left-5 top-3"/>
                    </div>
 
                </div>
                <div className="position-absolute">
                    <div className={this.state.calendarContentCss} >
                        <CalendarContent
                            titleMode="day"
                            directionBtnMode="month"
                            switch={false}
                            dateSelect={this.dateSelect.bind(this)}
                            changeInputDate={this.changeInputDate.bind(this)}
                            changeCurrentCalendar={this.formatDatePickerToNumber()}
                        />
 
                    </div>
                </div>
 
 
            </div>
        )
    }
}
 
export default DatePicker;
