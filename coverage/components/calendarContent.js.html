import React, { Component } from "react";
import _ from 'lodash';
 
export class CalendarContent extends Component{
    constructor(){
        super();
        let today = this.getToday();
        this.modeType=['day','month','year'];
 
        this.state ={
            currentCalendar:{year:today.year, month:today.month, day:today.day},
            mode:0
        };
    }
    componentDidMount(){
 
    }
 
    componentWillReceiveProps(nextProps){
        this.setStateIfPropsChange(nextProps);
    }
 
    setStateIfPropsChange(nextProps){
        if(nextProps.changeCurrentCalendar!==this.props.changeCurrentCalendar){
            this.setCurrentCalendar(nextProps.changeCurrentCalendar);
        }
    }
 
    setCurrentCalendar(changeCurrentCalendar){
        this.setState({currentCalendar: _.clone(changeCurrentCalendar) });
    }
 
    getToday(){
        let date = new Date();
        let year = date.getFullYear(),
            month = date.getMonth()+1,
            day = date.getDate();
 
        let today={'year':year,'month':month,'day':day};
 
        return today;
    }
 
    renderTitle(){
 
        let currentCalendar = this.state.currentCalendar;
        let month = ['Jan', 'Feb', 'Mar', 'Apr', 'May', 'Jun', 'Jul', 'Aug', 'Sep', 'Oct', 'Nov', 'Dec'];
        let mode = this.props.titleMode==undefined ? this.modeType[this.state.mode]:this.props.titleMode ;
        let title;
        switch (mode){
            case 'day':
                title =
                    <div >
                        <span>{month[currentCalendar.month-1]}</span>
                        <span className="ml-3">{currentCalendar.year}</span>
                    </div>;
                break;
            case 'month':
                title =
                    <div >
                        <span>{currentCalendar.year}</span>
                    </div>;
                break;
            case 'year':
                let beginYear = currentCalendar.year - currentCalendar.year%10;
                let lastYear = beginYear+9;
 
                title =
                    <div >
                        <span>{beginYear}-{lastYear}</span>
                    </div>;
                break;
        }
 
        if(this.props.switch==undefined || this.props.switch==true){
            title =
                <div onClick={this.switchMode.bind(this,'next')}>
                    {title}
                </div>
        }
 
        return(
            <div className="d-flex justify-content-around font-weight-bold">
                <i className="fas fa-chevron-left" onClick={this.directionBtn.bind(this,'previous')}/>
 
 
                {title}
 
                <i className="fas fa-chevron-right" onClick={this.directionBtn.bind(this,'next')}/>
            </div>
        );
    }
 
    directionBtn(direction){
        let mode = this.props.directionBtnMode==undefined ? this.modeType[this.state.mode]:this.props.directionBtnMode ;
 
 
 
        let currentCalendar = this.state.currentCalendar;
        switch (mode){
            case 'day':
                if(direction=='previous'){
                    if(currentCalendar.day==1){
                        if(currentCalendar.month==1){
                            currentCalendar.year --;
                            currentCalendar.month =12;
                        }else {
                            currentCalendar.month --;
                        }
 
                        currentCalendar.day = this.getLengthOfMonth(currentCalendar.year,currentCalendar.month);
 
                    }else{
                        currentCalendar.day --;
                    }
                }else{
                    let monthLength = this.getLengthOfMonth(currentCalendar.year,currentCalendar.month);
 
                    if(currentCalendar.day==monthLength){
                        if(currentCalendar.month==12){
                            currentCalendar.year ++;
                            currentCalendar.month = 1;
                        }else {
                            currentCalendar.month ++;
                        }
                        currentCalendar.day = 1;
                    }else{
                        currentCalendar.day ++;
                    }
                }
 
                break;
            case 'month':
                if(direction=='previous'){
                    if(currentCalendar.month==1){
                        currentCalendar.year --;
                        currentCalendar.month =12;
                    }else{
                        currentCalendar.month --;
                    }
                }else{
                    if(currentCalendar.month==12){
                        currentCalendar.year ++;
                        currentCalendar.month = 1;
                    }else{
                        currentCalendar.month ++;
                    }
                }
 
                break;
            case 'year':
                if(direction=='previous'){
                    currentCalendar.year --;
                }else{
                    currentCalendar.year ++;
                }
 
                break;
        }
 
        this.setState({currentCalendar});
    }
 
    switchMode(direction){
        let mode = this.state.mode;
 
        if(direction=='next'){
            if(mode<2){
                mode++;
            }
        }else {
            if(mode>0){
                mode--;
            }
        }
 
        this.setState({mode});
 
    }
 
    renderTable(){
        return(
            <table>
                {this.setThead(this.modeType[this.state.mode])}
                {this.setTbody(this.modeType[this.state.mode])}
            </table>
        );
    }
 
    setThead(mode){
 
        if(mode=='day'){
            let weekdays = ['Su', 'Mo', 'Tu', 'We', 'Th', 'Fr', 'Sa'];
            let tHead=[];
            for(let counter=0; counter<weekdays.length; counter++){
                tHead.push(
                    <th className="calendar-md text-center " key={counter}>{weekdays[counter]}</th>
                );
            }
 
            return(
                <thead>
                <tr>
                    {tHead}
                </tr>
                </thead>
            );
        }
 
    }
 
    setTbody(mode){
        let calendar;
        let getCalendarFormat;
        let currentCalendar = this.state.currentCalendar;
 
        switch(mode) {
            case 'day':
                calendar = this.calculateDayCalendar(currentCalendar.year, currentCalendar.month);
                getCalendarFormat = this.calendarFormat(7,calendar);
                break;
            case 'month':
                calendar = this.calculateMonthCalendar();
                getCalendarFormat = this.calendarFormat(4,calendar);
                break;
            case 'year':
                calendar = this.calculateYearCalendar(currentCalendar.year);
                getCalendarFormat = this.calendarFormat(4,calendar);
                break;
        }
 
        return(
            <tbody>
            {getCalendarFormat}
            </tbody>
        );
    }
 
    calculateDayCalendar(year, month){
 
        let calendarLength = 42;
        let dateCollection=[];
        let firstDayOfMonth = this.converseDateIntoWeek(year, month, 1);
 
        let currentMonthLength = this.getLengthOfMonth(year,month);
 
        let passMonthLength;
 
        if(month==1){
            passMonthLength = this.getLengthOfMonth(year-1,12);
        }else{
            passMonthLength = this.getLengthOfMonth(year,month-1);
        }
 
 
        let futureMonthDay =1;
 
        for(let counter=0; counter<calendarLength; counter++){
            if(counter<firstDayOfMonth){
                let passMonthDay = passMonthLength-firstDayOfMonth+counter+1;
                dateCollection.push({status:'pass',date:passMonthDay});
            }else if(counter>=firstDayOfMonth && counter < (firstDayOfMonth+currentMonthLength) ){
                let currentMonthDay = counter-firstDayOfMonth+1;
                dateCollection.push({status:'current',date:currentMonthDay});
            }else {
                dateCollection.push({status:'future',date:futureMonthDay});
                futureMonthDay++;
            }
        }
 
        return dateCollection;
    }
 
    converseDateIntoWeek(year,month,day){
 
        let monthFormat = month,
            yearFormat = year;
 
        if(month<=2){
            monthFormat += 12;
            yearFormat --;
        }
 
        let dateFormat = (yearFormat-1) + Math.floor((yearFormat-1)/4) - Math.floor((yearFormat-1)/100) + Math.floor((yearFormat-1)/400) + Math.floor( 13 * (monthFormat+1) / 5 ) + day;
 
        let dayOfWeek = dateFormat%7;
        return dayOfWeek;
    }
 
    getLengthOfMonth(year,month){
        let monthLength = 30;
 
        if(month==2){
            if((year % 4 == 0 && year % 100 != 0) || (year % 400 == 0 )){
                monthLength = 29;
            }else {
                monthLength = 28;
            }
        }else if(month==7||month==8){
            monthLength = 31;
        }else if(month<7 && month%2==1){
            monthLength = 31;
        }else if(month>8 && month%2==0){
            monthLength = 31;
        }
 
        return monthLength;
    }
 
    calculateMonthCalendar(){
        let month = ['Jan', 'Feb', 'Mar', 'Apr', 'May', 'Jun', 'Jul', 'Aug', 'Sep', 'Oct', 'Nov', 'Dec'];
 
        let monthFormat = month.map(eachMonth=>{
            return {status:'current',date:eachMonth};
        });
 
        return monthFormat;
    }
 
    calculateYearCalendar(year){
        let calendarLength =12;
        let firstColumnOfCurrent = 1;
        let lastColumnOfCurrent = 10;
        let dateCollection=[];
 
        let beginYear = year - year%10 - 1;
 
 
        for(let counter=0; counter<calendarLength; counter++){
            if(counter<firstColumnOfCurrent){
                dateCollection.push({status:'pass',date:beginYear+counter});
            }else if(counter>=firstColumnOfCurrent && counter <= lastColumnOfCurrent){
                dateCollection.push({status:'current',date:beginYear+counter});
            }else {
                dateCollection.push({status:'future',date:beginYear+counter});
            }
        }
 
        return dateCollection;
    }
 
 
    calendarFormat(column,calendar){
 
        let rowLength = calendar.length/column;
 
        let rowCollection=[];
 
        for(let rowCounter=0; rowCounter<rowLength; rowCounter++){
 
            let columnCollection=[];
            for(let columnCounter=0; columnCounter<column; columnCounter++){
 
                let calendarCounter = rowCounter*column+columnCounter;
                let date = calendar[calendarCounter].date;
                let status = calendar[calendarCounter].status;
                let dateToSelectDate = date;
                let mode = this.modeType[this.state.mode];
                if(mode=='month'){
                    dateToSelectDate = calendarCounter+1;
                }
 
                let cssName = this.getDateCss(status,date);
 
                columnCollection.push(
                    <td key={columnCounter} onClick={this.selectDate.bind(this,status,dateToSelectDate)} onDoubleClick={this.switchMode.bind(this,'previous')}>
                        {/*<td key={columnCounter}  >*/}
                        <span className={cssName}>{date}</span>
                    </td>
                );
 
            }
 
            rowCollection.push(
                <tr key={rowCounter} className="">
                    {columnCollection}
                </tr>
            );
 
        }
 
        return rowCollection;
    }
 
    getDateCss(status,eachCalendar){
        let today = this.getToday();
        let currentCalendar = this.state.currentCalendar;
        let mode = this.modeType[this.state.mode];
        let dateSelect = this.props.dateSelect(mode,eachCalendar,currentCalendar,today);
 
        let cssName ="calendar-md d-flex justify-content-center align-items-center ";
        if(status!='current'){
            cssName += 'text-gray-100';
        }else{
            if(dateSelect.isSelect ){
                cssName += 'border-radius-8 bg-danger text-white';
            }else if(dateSelect.isToday){
                cssName += 'text-danger';
            }
        }
 
        return cssName;
    }
 
    selectDate(status,date){
 
        let mode = this.modeType[this.state.mode];
        let currentCalendar = this.state.currentCalendar;
 
        switch(mode){
            case 'day':
                if(status=='pass'){
                    if(currentCalendar.month==1){
                        currentCalendar.year --;
                        currentCalendar.month =12;
                    }else {
                        currentCalendar.month --;
                    }
                    currentCalendar.day = date;
                }else if(status=='future'){
                    if(currentCalendar.month==12){
                        currentCalendar.year ++;
                        currentCalendar.month = 1;
                    }else {
                        currentCalendar.month ++;
                    }
                    currentCalendar.day = date;
                }else{
                    currentCalendar.day = date;
                }
 
                break;
 
            case 'month':
                currentCalendar.month=date;
                break;
 
            case 'year':
                currentCalendar.year=date;
                break;
        }
 
        if(this.props.changeInputDate){
            this.props.changeInputDate(_.clone(currentCalendar));
        }
 
        this.setState({currentCalendar});
    }
 
 
 
    render(){
        return(
            <div className="border px-3 py-4">
                {this.renderTitle()}
                {this.renderTable()}
            </div>
        )
    }
}
 
export default CalendarContent;
